#jQuery日期选择器 jquery.datePicker

默认: 
$('#date-range0').datePicker();
默认(自动关闭): 
$('#date-range0_0').datePicker({
	autoClose: true//自动关闭，默认为false
});
小时: 
$('#date-range1').datePicker({
	format: 'YYYY/MM/DD HH:mm',//日期格式，默认为YYYY/MM/DD #更多格式参考 http://momentjs.com/docs/#/displaying/format/
	time: {
		enabled: true//时间显示，默认为false
	}
});
范围: 
$('#date-range2').datePicker({
	rangeDate:true//日期范围选择，默认为false
});
范围+小时: 
$('#date-range3').datePicker({
	rangeDate: true,
	format: 'YYYY/MM/DD HH:mm',
	time: {
		enabled: true
	}
});
范围+小时(单日期框): 
$('#date-range4').datePicker({
	singleMonth: true,//单日期框，范围选择时默认为false
	rangeDate:true,
	format: 'YYYY/MM/DD HH:mm',
	time: {
		enabled: true
	}
});
范围(单日期框): 
$('#date-range5').datePicker({
	singleMonth: true,
	rangeDate:true,
});
范围(指定区间): 
$('#date-range6').datePicker({
	rangeDate:true,
	startDate:['today',-10],//起始时间，格式为时间戳，或者['today',天数]
	endDate:['today',10],//截止时间，格式为时间戳，或者['today',天数]
	maxDays:20,//最长时间段
	minDays:7//最短时间段
});
